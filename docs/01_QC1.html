add_qc_percentage_to_metadata <- function(seurat_objects_list) {
  lapply(seurat_objects_list, function(seurat_obj) {
    seurat_obj <- subset(seurat_obj,
                         subset = nCount_RNA > 0)
    
    seurat_obj[["percent.mt"]] <- PercentageFeatureSet(seurat_obj, 
                                                       pattern = "^MT-")
    seurat_obj[["percent.hb"]] <- PercentageFeatureSet(seurat_obj, 
                                                       pattern = "^HB[^(P)]")
    seurat_obj[["percent.ribo"]] <- PercentageFeatureSet(seurat_obj,
                                                         pattern = "^RPL|^RPS")
    seurat_obj
  })
}

DoubletFinder Preparation

Before running DoubletFinder we need to normalize, scale, find variable genes and run PCA. Here I chose approx = FALSE for RunPCA to make the run independent of what processor we are using (deterministic).

set.seed(123)
process_seurat_objects <- function(seurat_object_list) {
  
  lapply(seq_along(seurat_object_list), function(i) {
    seurat_obj  <- seurat_object_list[[i]]
    sample_name <- names(seurat_object_list)[i]
    
    # Step 1: Normalize (scale factor = median nCount_RNA of this sample)
    scale_factor <- median(seurat_obj$nCount_RNA)
    print(scale_factor)
    seurat_obj <- NormalizeData(seurat_obj, 
                                scale.factor = scale_factor)
    hist(seurat_obj[["nCount_RNA"]]$nCount_RNA, 
         breaks = 20, 
         main = sample_name)
    
    # Step 2: Variable features
    seurat_obj <- FindVariableFeatures(seurat_obj, 
                                       selection.method = "vst", 
                                       nfeatures = 2000)
    
    # Step 3: Scale on all genes
    seurat_obj <- ScaleData(seurat_obj, 
                            features = rownames(seurat_obj), 
                            verbose = FALSE)
    
    # Step 4: PCA (approx = FALSE for deterministic, machine-independent results)
    seurat_obj <- RunPCA(seurat_obj,
                         features = VariableFeatures(seurat_obj),
                         approx   = FALSE,
                         verbose  = FALSE)
    
    # Step 5: Elbow plot
    print(ElbowPlot(seurat_obj, ndims = 50, reduction = "pca") + ggtitle(sample_name))
    
    seurat_obj
  }) |> setNames(names(seurat_object_list))
}

process_seurat_clustering <- function(seurat_object_list, dims = 1:35, resolution = 0.3) {
  
  lapply(seq_along(seurat_object_list), function(i) {
    set.seed(123) # re-seeded per sample for reproducibility across each individual sample
    seurat_obj  <- seurat_object_list[[i]]
    sample_name <- names(seurat_object_list)[i]
    
    # Step 1: Neighbors
    seurat_obj <- FindNeighbors(seurat_obj, dims = dims, verbose = FALSE)
    
    # Step 2: Clustering
    seurat_obj <- FindClusters(seurat_obj, resolution = resolution, verbose = FALSE)
    
    # Step 3: UMAP
    seurat_obj <- RunUMAP(seurat_obj, dims = dims, verbose = FALSE)
    
    seurat_obj
  }) |> setNames(names(seurat_object_list))
}

pN-pK parameter sweep

We are going to calculate the optimal pK value per sample. This pK value is the neighborhood size DoubletFinder will use to calculate the fraction of artificial doublets in a neighborhood (pANN value). Samples with high pANN values sit in regions that are “doublet-like” and are identified as doublets. The size of the neighborhood pK will influence the pANN value.

pN is the amount of artificial doublets added but DoubletFinder is largely insensitive to this value. summarizeSweep() will compute the bimodality of the paNN distribution for different pN-pK value pairs. Ideally the pANN distribution would be strictly bimodal, so that singlets are grouped together and doublets are grouped together. find.pK() then identifies which pK leads to the best separation. It might be arbitrary for example if the BCmetric curve is flat or noisy and that is why we are also computing plots to look at. Ideally the plots have an obvious single peak.

set.seed(123)

process_seurat_sweep <- function(seurat_object_list,
                                 output_directory = plots_output_directory,
                                 PCs = 1:35) {
  
  lapply(seq_along(seurat_object_list), function(i) {
    set.seed(123)                                   # re-seeded per sample for reproducibility
    seurat_obj  <- seurat_object_list[[i]]
    sample_name <- names(seurat_object_list)[i]
    
    # Sweep pN (proportion of artificial doublets) x pK (kNN neighbourhood size)
    # Generates artificial doublets and computes pANN for every parameter pair
    sweep.res.list <- paramSweep(seurat_obj, PCs = PCs, sct = FALSE)
    
    # For each pN-pK pair, compute the bimodality coefficient of the pANN distribution
    sweep.stats <- summarizeSweep(sweep.res.list, GT = FALSE)
    
    # BCmetric = mean-variance-normalised BC across pN; its max marks the pK
    # where singlets and doublets separate most cleanly
    bcmvn <- find.pK(sweep.stats)
    
    # Optimal pK = the one maximising BCmetric
    pK <- bcmvn %>%
      filter(BCmetric == max(BCmetric)) %>%
      pull(pK) %>%
      as.character() %>%
      as.numeric()
    
    # Plot the BCmetric curve — inspect these: a good sweep has one clear peak,
    # a flat/noisy curve or a peak at the range edge means the argmax is unreliable
    sweep_plot <- ggplot(bcmvn, aes(pK, BCmetric, group = 1)) +
      geom_point() +
      geom_line() +
      geom_vline(xintercept = which.max(bcmvn$BCmetric),   # mark the selected pK
                 linetype = "dashed", colour = "red") +
      theme_minimal() +
      ggtitle(paste("Sweep results for", sample_name),
              subtitle = paste("Selected pK =", pK)) +
      xlab("pK") +
      ylab("BCmetric") +
      theme(axis.text.x = element_text(angle = 90, vjust = 0.5, size = 7))
    
    ggsave(filename = file.path(output_directory,
                                paste0("01_QC1_pKSweep_", sample_name, ".png")),
           plot = sweep_plot, width = 8, height = 6)
    
    pK
  }) |> setNames(names(seurat_object_list))
}

DF_list <- function(seurat_object_list, pK_list, DB_rate = 0.05, PCs = 1:35, pN = 0.25) {
  
  lapply(seq_along(seurat_object_list), function(i) {
    set.seed(123)
    seurat_obj  <- seurat_object_list[[i]]
    sample_name <- names(seurat_object_list)[i]
    
    # pK looked up by name, not position
    pK <- pK_list[[sample_name]]
    
    # Expected doublets: Poisson estimate, adjusted down for homotypic doublets
    # (doublets of two cells of the same type are undetectable by expression)
    annotations    <- as.character(seurat_obj@meta.data$seurat_clusters)
    homotypic.prop <- modelHomotypic(annotations)
    nExp_poi       <- round(DB_rate * ncol(seurat_obj))
    nExp_poi.adj   <- round(nExp_poi * (1 - homotypic.prop))
    
    seurat_obj <- doubletFinder(seurat_obj,
                                PCs        = PCs,
                                pN         = pN,
                                pK         = pK,
                                nExp       = nExp_poi.adj,
                                reuse.pANN = NULL,
                                sct        = FALSE)
    
    # Column name is built from the parameters used
    df_col <- paste0("DF.classifications_", pN, "_", pK, "_", nExp_poi.adj)
    
    p <- DimPlot(seurat_obj, 
                 reduction = "umap", 
                 group.by = df_col,
                 cols = c("Singlet" = "#015EA8", "Doublet" = "#B56676")) +
            ggtitle(sample_name)
    
    print(p)
    
    ggsave(filename = file.path(plots_output_directory,
                                paste0("01_QC1_DoubletDimPlot", sample_name, ".png")), 
           plot = p)
    
    seurat_obj$doublet <- seurat_obj[[df_col]]
    seurat_obj
  }) |> setNames(names(seurat_object_list))
}

ProportionPlot <- function(object, name1, name2, cols = NULL, as_percent = FALSE) {
  # Stacked bar plot: proportion of `name1` levels within each level of `name2`.
  # Each bar (one per name2 level) sums to 1.
  # Example: ProportionPlot(merged, "seurat_clusters", "sample")

  md <- object@meta.data
  stopifnot(all(c(name1, name2) %in% colnames(md)))

  prop_df <- table(md[[name1]], md[[name2]]) |>
    prop.table(margin = 2) |>   # normalise within each name2 level
    as.data.frame() |>
    setNames(c(name1, name2, "proportion"))

  p <- ggplot(prop_df, aes(x = .data[[name2]], y = proportion, fill = .data[[name1]])) +
    geom_col(colour = "black") +
    labs(x = name2, y = "Proportion of cells", fill = name1) +
    theme_linedraw() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1))

  if (!is.null(cols))  p <- p + scale_fill_manual(values = cols)
  if (as_percent)      p <- p + scale_y_continuous(labels = scales::percent)

  p
  
}

  combined_data <- data.frame()
  
  for (sample_name in names(seurat_objects_list)) {
    # Access the Seurat object
    seurat_obj <- seurat_objects_list[[sample_name]]
    
    # Check if the required metadata columns exist
    if (!all(c("nFeature_RNA", "nCount_RNA", "doublet") %in% colnames(seurat_obj@meta.data))) {
      stop("The Seurat object must contain 'nFeature_RNA', 'nCount_RNA', and 'doublet' in the metadata.")
    }
    
    # Extract the data for plotting
    plot_data <- seurat_obj@meta.data[, c("nFeature_RNA", "nCount_RNA", "doublet")]
    plot_data$sample <- sample_name
    
    # Combine the data
    combined_data <- rbind(combined_data, plot_data)
  }
  
  # Create the scatter plot
  p <- ggplot(combined_data, aes(x = nCount_RNA, y = nFeature_RNA, color = doublet, label = sample)) +
    geom_point(size = 0.5) +
    scale_color_manual(values = c("Singlet" = "#015EA8", "Doublet" = "#B56676")) +
    labs(
      x = "nCount_RNA",
      y = "nFeature_RNA",
      color = "Doublet Class"
    ) +
    ggtitle("Scatter plot of nFeature_RNA vs. nCount_RNA for All Samples") +
    theme_minimal() +
    facet_wrap(~sample, scales = "free")
  
  # Print the plot
  print(p)
  file_path <- file.path(plots_output_directory, paste0("DoubletFinder_", "scatterplot", ".png"))
